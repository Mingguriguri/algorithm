m = int(input())
cards = [int(x) for x in input().split()]

answer = [0] * m
cards.sort()       

for i in range(len(sk)):
    target = sk[i]
    left = 0
    right = cards[-1]

    while left <= right:
        if left < target:
            left += 1
        elif right >= target:
            right -= 1
        elif left == target:
            answer[i] += 1
            left += 1
        elif right == target:
            answer[i] += 1
            right -= 1

for i in range(len(answer)):
    print(answer[i], end= ' ')
