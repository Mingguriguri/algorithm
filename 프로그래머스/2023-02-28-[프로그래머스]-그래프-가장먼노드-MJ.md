```python
def BFS_with_adj_list(graph, root):
    visited = []
    queue = deque([root])

    while queue:
        n = queue.popleft()

            visited.append(n)
            queue += graph[n] - set(visited)
    return visited
```

```python
from collections import deque

def solution(n, edge):
    graph = [[] for _ in range(n+1) ]
    visited = [0] * (n+1)
    queue = deque()
    queue.append(1)
    visited[1] = 1
    nodes = 0
    
    for a, b in edge:
        graph[a].append(b)
        graph[b].append(a)
    
    while queue:
        nodes = len(queue)
        for i in range(nodes):
            cur_n = queue.popleft()
            for c in graph[cur_n]:
                if visited[c] == 0:
                    visited[c] = 1
                    queue.append(c)
        
    return nodes
```
